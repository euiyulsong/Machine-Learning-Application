def rotation_matrix(A, alpha):
    alpha = np.radians(alpha)
    return A.dot(np.matrix([[np.cos(alpha), -np.sin(alpha)], [np.sin(alpha), np.cos(alpha)]]))
B = np.matrix([[0, 0], [0, 1], [0.5, 0.5]])

plt.plot(rotation_matrix(B, 0))
plt.plot(rotation_matrix(B, 45))
plt.plot(rotation_matrix(B, 90))
plt.plot(rotation_matrix(B, -225))
print("")
